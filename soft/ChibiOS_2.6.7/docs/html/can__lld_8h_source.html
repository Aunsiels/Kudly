176  /**
177  * @brief One or more frames become available.
178  * @note After broadcasting this event it will not be broadcasted again
179  * until the received frames queue has been completely emptied. It
180  * is <b>not</b> broadcasted for each received frame. It is
181  * responsibility of the application to empty the queue by
182  * repeatedly invoking @p chReceive() when listening to this event.
183  * This behavior minimizes the interrupt served by the system
184  * because CAN traffic.
185  * @note The flags associated to the listeners will indicate which
186  * receive mailboxes become non-empty.
187  */
188  EventSource rxfull_event;
189  /**
190  * @brief One or more transmission mailbox become available.
191  * @note The flags associated to the listeners will indicate which
192  * transmit mailboxes become empty.
193  *
194  */
195  EventSource txempty_event;
196  /**
197  * @brief A CAN bus error happened.
198  * @note The flags associated to the listeners will indicate the
199  * error(s) that have occurred.
