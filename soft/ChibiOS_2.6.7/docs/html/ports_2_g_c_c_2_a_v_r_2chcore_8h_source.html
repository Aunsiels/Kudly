151 /**
152  * @brief Platform dependent part of the @p chThdCreateI() API.
153  * @details This code usually setup the context switching frame represented
154  * by an @p intctx structure.
155  */
156 #define SETUP_CONTEXT(workspace, wsize, pf, arg) { \
157  tp->p_ctx.sp = (struct intctx*)((uint8_t *)workspace + wsize - \
158  sizeof(struct intctx)); \
159  tp->p_ctx.sp->r2 = (int)pf; \
160  tp->p_ctx.sp->r3 = (int)pf >> 8; \
161  tp->p_ctx.sp->r4 = (int)arg; \
162  tp->p_ctx.sp->r5 = (int)arg >> 8; \
163  tp->p_ctx.sp->pcl = (int)_port_thread_start >> 8; \
164  tp->p_ctx.sp->pch = (int)_port_thread_start; \
165 }
