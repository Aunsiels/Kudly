23  * File: @ref testqueues.c
24  *
25  * <h2>Description</h2>
26  * This module implements the test sequence for the @ref io_queues subsystem.
27  * The tests are performed by inserting and removing data from queues and by
28  * checking both the queues status and the correct sequence of the extracted
29  * data.
30  *
31  * <h2>Objective</h2>
32  * Objective of the test module is to cover 100% of the @ref io_queues code.<br>
33  * Note that the @ref io_queues subsystem depends on the @ref semaphores
34  * subsystem that has to met its testing objectives as well.
35  *
36  * <h2>Preconditions</h2>
37  * The module requires the following kernel options:
38  * - @p CH_USE_QUEUES (and dependent options)
39  * .
40  * In case some of the required options are not enabled then some or all tests
41  * may be skipped.
42  *
43  * <h2>Test Cases</h2>
44  * - @subpage test_queues_001
45  * - @subpage test_queues_002
46  * .
47  * @file testqueues.c

90 static void queues1_execute(void) {
91  unsigned i;
92  size_t n;
93 
94  /* Initial empty state */
95  test_assert_lock(1, chIQIsEmptyI(&iq), "not empty");
96 
97  /* Queue filling */
98  chSysLock();
99  for (i = 0; i < TEST_QUEUES_SIZE; i++)
100  chIQPutI(&iq, 'A' + i);
101  chSysUnlock();
102  test_assert_lock(2, chIQIsFullI(&iq), "still has space");
103  test_assert_lock(3, chIQPutI(&iq, 0) == Q_FULL, "failed to report Q_FULL");
104 
105  /* Queue emptying */
106  for (i = 0; i < TEST_QUEUES_SIZE; i++)
107  test_emit_token(chIQGet(&iq));
108  test_assert_lock(4, chIQIsEmptyI(&iq), "still full");
109  test_assert_sequence(5, "ABCD");
110 
111  /* Queue filling again */
112  chSysLock();
113  for (i = 0; i < TEST_QUEUES_SIZE; i++)
114  chIQPutI(&iq, 'A' + i);
115  chSysUnlock();
116 
117  /* Reading the whole thing */
118  n = chIQReadTimeout(&iq, wa[1], TEST_QUEUES_SIZE * 2, TIME_IMMEDIATE);
119  test_assert(6, n == TEST_QUEUES_SIZE, "wrong returned size");
120  test_assert_lock(7, chIQIsEmptyI(&iq), "still full");
121 
122  /* Queue filling again */
123  chSysLock();
124  for (i = 0; i < TEST_QUEUES_SIZE; i++)
125  chIQPutI(&iq, 'A' + i);
126  chSysUnlock();
127 
128  /* Partial reads */
129  n = chIQReadTimeout(&iq, wa[1], TEST_QUEUES_SIZE / 2, TIME_IMMEDIATE);
130  test_assert(8, n == TEST_QUEUES_SIZE / 2, "wrong returned size");
131  n = chIQReadTimeout(&iq, wa[1], TEST_QUEUES_SIZE / 2, TIME_IMMEDIATE);
132  test_assert(9, n == TEST_QUEUES_SIZE / 2, "wrong returned size");
133  test_assert_lock(10, chIQIsEmptyI(&iq), "still full");
134 
135  /* Testing reset */
136  chSysLock();
137  chIQPutI(&iq, 0);
138  chIQResetI(&iq);
139  chSysUnlock();
140  test_assert_lock(11, chIQGetFullI(&iq) == 0, "still full");
141  threads[0] = chThdCreateStatic(wa[0], WA_SIZE, chThdGetPriority()+1, thread1, NULL);
142  test_assert_lock(12, chIQGetFullI(&iq) == 0, "not empty");
143  test_wait_threads();
144 
145  /* Timeout */
146  test_assert(13, chIQGetTimeout(&iq, 10) == Q_TIMEOUT, "wrong timeout return");
147 }

177 static void queues2_execute(void) {

181  /* Initial empty state */

183 
184  /* Queue filling */
185  for (i = 0; i < TEST_QUEUES_SIZE; i++)
186  chOQPut(&oq, 'A' + i);
187  test_assert_lock(2, chOQIsFullI(&oq), "still has space");
188 
189  /* Queue emptying */
190  for (i = 0; i < TEST_QUEUES_SIZE; i++) {
191  char c;
192 
193  chSysLock();
194  c = chOQGetI(&oq);
195  chSysUnlock();
196  test_emit_token(c);
197  }
198  test_assert_lock(3, chOQIsEmptyI(&oq), "still full");
199  test_assert_sequence(4, "ABCD");
200  test_assert_lock(5, chOQGetI(&oq) == Q_EMPTY, "failed to report Q_EMPTY");
201 
202  /* Writing the whole thing */
203  n = chOQWriteTimeout(&oq, wa[1], TEST_QUEUES_SIZE * 2, TIME_IMMEDIATE);
204  test_assert(6, n == TEST_QUEUES_SIZE, "wrong returned size");
205  test_assert_lock(7, chOQIsFullI(&oq), "not full");
206  threads[0] = chThdCreateStatic(wa[0], WA_SIZE, chThdGetPriority()+1, thread2, NULL);
207  test_assert_lock(8, chOQGetFullI(&oq) == TEST_QUEUES_SIZE, "not empty");
208  test_wait_threads();
209 
210  /* Testing reset */
211  chSysLock();
212  chOQResetI(&oq);
213  chSysUnlock();
214  test_assert_lock(9, chOQGetFullI(&oq) == 0, "still full");
215 
216  /* Partial writes */
217  n = chOQWriteTimeout(&oq, wa[1], TEST_QUEUES_SIZE / 2, TIME_IMMEDIATE);
218  test_assert(10, n == TEST_QUEUES_SIZE / 2, "wrong returned size");
219  n = chOQWriteTimeout(&oq, wa[1], TEST_QUEUES_SIZE / 2, TIME_IMMEDIATE);
220  test_assert(11, n == TEST_QUEUES_SIZE / 2, "wrong returned size");
221  test_assert_lock(12, chOQIsFullI(&oq), "not full");
222 
223  /* Timeout */
224  test_assert(13, chOQPutTimeout(&oq, 0, 10) == Q_TIMEOUT, "wrong timeout return");
