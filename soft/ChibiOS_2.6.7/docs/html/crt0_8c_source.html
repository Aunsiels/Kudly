57 /*

60  */

62  uint32_t *p1 = start; \
63  uint32_t *p2 = end; \
64  while (p1 < p2) \

262 /**
263  * @brief Reset vector.
264  */
265 #if !defined(__DOXYGEN__)
266 __attribute__((naked))
267 #endif
268 void ResetHandler(void) {
269  uint32_t psp, reg;
270 
271  /* Process Stack initialization, it is allocated starting from the
272  symbol __process_stack_end__ and its lower limit is the symbol
273  __process_stack_base__.*/
274  asm volatile ("cpsid i");
275  psp = SYMVAL(__process_stack_end__);
276  asm volatile ("msr PSP, %0" : : "r" (psp));
277 
278 #if CORTEX_USE_FPU
279  /* Initializing the FPU context save in lazy mode.*/
280  SCB_FPCCR = FPCCR_ASPEN | FPCCR_LSPEN;
281 
282  /* CP10 and CP11 set to full access.*/
283  SCB_CPACR |= 0x00F00000;
284 
285  /* FPSCR and FPDSCR initially zero.*/
286  reg = 0;
287  asm volatile ("vmsr FPSCR, %0" : : "r" (reg) : "memory");
288  SCB_FPDSCR = reg;
289 
290  /* CPU mode initialization, enforced FPCA bit.*/
291  reg = CRT0_CONTROL_INIT | 4;
292 #else
293  /* CPU mode initialization.*/
294  reg = CRT0_CONTROL_INIT;
295 #endif
296  asm volatile ("msr CONTROL, %0" : : "r" (reg));
297  asm volatile ("isb");
298 
299 #if CRT0_INIT_STACKS
300  /* Main and Process stacks initialization.*/
301  fill32(&__main_stack_base__,
302  &__main_stack_end__,
303  CRT0_STACKS_FILL_PATTERN);
304  fill32(&__process_stack_base__,
305  &__process_stack_end__,
306  CRT0_STACKS_FILL_PATTERN);
307 #endif
308 
309  /* Early initialization hook invocation.*/
310  __early_init();
311 
312 #if CRT0_INIT_DATA
313  /* DATA segment initialization.*/
314  {
315  uint32_t *tp, *dp;
316 
317  tp = &_textdata;
318  dp = &_data;
319  while (dp < &_edata)
320  *dp++ = *tp++;
321  }
322 #endif
323 
324 #if CRT0_INIT_BSS
325  /* BSS segment initialization.*/
326  fill32(&_bss_start, &_bss_end, 0);
327 #endif
328 
329  /* Late initialization hook invocation.*/
330  __late_init();
331 
332 #if CRT0_CALL_CONSTRUCTORS
333  /* Constructors invocation.*/
334  {
335  funcpp_t fpp = &__init_array_start;
336  while (fpp < &__init_array_end) {
337  (*fpp)();
338  fpp++;
339  }
340  }
341 #endif
342 
343  /* Invoking application main() function.*/
344  main();
345 
346 #if CRT0_CALL_DESTRUCTORS
347  /* Destructors invocation.*/
348  {
349  funcpp_t fpp = &__fini_array_start;
350  while (fpp < &__fini_array_end) {
351  (*fpp)();
352  fpp++;
353  }
354  }
355 #endif
356 
357  /* Invoking the exit handler.*/
358  _default_exit();
359 }
