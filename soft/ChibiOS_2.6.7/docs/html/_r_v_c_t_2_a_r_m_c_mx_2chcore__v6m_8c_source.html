99 /**
100  * @brief IRQ epilogue code.
101  *
102  * @param[in] lr value of the @p LR register on ISR entry
103  */
104 void _port_irq_epilogue(regarm_t lr) {
105 
106  if (lr != (regarm_t)0xFFFFFFF1) {
107  register struct extctx *ctxp;
108  register uint32_t psp __asm("psp");
109 
110  port_lock_from_isr();
111  /* Adding an artificial exception return context, there is no need to
112  populate it fully.*/
113  ctxp = (struct extctx *)psp;
114  ctxp--;
115  psp = (uint32_t)ctxp;
116  ctxp->xpsr = (regarm_t)0x01000000;
117 
118  /* The exit sequence is different depending on if a preemption is
119  required or not.*/
120  if (chSchIsPreemptionRequired()) {
121  /* Preemption is required we need to enforce a context switch.*/
122  ctxp->pc = (regarm_t)_port_switch_from_isr;
123  }
124  else {
125  /* Preemption not required, we just need to exit the exception
126  atomically.*/
127  ctxp->pc = (regarm_t)_port_exit_from_isr;
128  }
129 
130  /* Note, returning without unlocking is intentional, this is done in
131  order to keep the rest of the context switch atomic.*/
132  }
133 }
