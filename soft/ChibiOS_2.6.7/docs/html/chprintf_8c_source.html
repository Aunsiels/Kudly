37 static char *long_to_string_with_divisor(char *p,
38  long num,
39  unsigned radix,
40  long divisor) {
41  int i;
42  char *q;
43  long l, ll;
44 
45  l = num;
46  if (divisor == 0) {
47  ll = num;
48  } else {
49  ll = divisor;
50  }
51 
52  q = p + MAX_FILLER;
53  do {
54  i = (int)(l % radix);
55  i += '0';
56  if (i > '9')
57  i += 'A' - '0' - 10;
58  *--q = i;
59  l /= radix;
60  } while ((ll /= radix) != 0);
61 
62  i = (int)(p + MAX_FILLER - q);
63  do
64  *p++ = *q++;
65  while (--i);
66 
67  return p;
68 }

88 /**
89  * @brief System formatted output function.
90  * @details This function implements a minimal @p vprintf()-like functionality
91  * with output on a @p BaseSequentialStream.
92  * The general parameters format is: %[-][width|*][.precision|*][l|L]p.
93  * The following parameter types (p) are supported:
94  * - <b>x</b> hexadecimal integer.
95  * - <b>X</b> hexadecimal long.
96  * - <b>o</b> octal integer.
97  * - <b>O</b> octal long.
98  * - <b>d</b> decimal signed integer.
99  * - <b>D</b> decimal signed long.
100  * - <b>u</b> decimal unsigned integer.
101  * - <b>U</b> decimal unsigned long.
102  * - <b>c</b> character.
103  * - <b>s</b> string.
104  * .
105  *
106  * @param[in] chp pointer to a @p BaseSequentialStream implementing object
107  * @param[in] fmt formatting string
108  * @param[in] ap list of parameters
109  *
110  * @api
111  */
112 void chvprintf(BaseSequentialStream *chp, const char *fmt, va_list ap) {
113  char *p, *s, c, filler;
114  int i, precision, width;
115  bool_t is_long, left_align;
116  long l;
117 #if CHPRINTF_USE_FLOAT
118  float f;
119  char tmpbuf[2*MAX_FILLER + 1];
120 #else
121  char tmpbuf[MAX_FILLER + 1];
122 #endif
123 
124  while (TRUE) {
125  c = *fmt++;
126  if (c == 0)
127  return;
128  if (c != '%') {
129  chSequentialStreamPut(chp, (uint8_t)c);
130  continue;
131  }
132  p = tmpbuf;
133  s = tmpbuf;
134  left_align = FALSE;
135  if (*fmt == '-') {
136  fmt++;
137  left_align = TRUE;
138  }
139  filler = ' ';
140  if ((*fmt == '.') || (*fmt == '0')) {
141  fmt++;
142  filler = '0';
143  }
144  width = 0;
145  while (TRUE) {
146  c = *fmt++;
147  if (c >= '0' && c <= '9')
148  c -= '0';
149  else if (c == '*')
150  c = va_arg(ap, int);
151  else
152  break;
153  width = width * 10 + c;
154  }
155  precision = 0;
156  if (c == '.') {
157  while (TRUE) {
158  c = *fmt++;
159  if (c >= '0' && c <= '9')
160  c -= '0';
161  else if (c == '*')
162  c = va_arg(ap, int);
163  else
164  break;
165  precision *= 10;
166  precision += c;
167  }
168  }
169  /* Long modifier.*/
170  if (c == 'l' || c == 'L') {
171  is_long = TRUE;
172  if (*fmt)
173  c = *fmt++;
174  }
175  else
176  is_long = (c >= 'A') && (c <= 'Z');
177 
178  /* Command decoding.*/
179  switch (c) {
180  case 'c':
181  filler = ' ';
182  *p++ = va_arg(ap, int);
183  break;
184  case 's':
185  filler = ' ';
186  if ((s = va_arg(ap, char *)) == 0)
187  s = "(null)";
188  if (precision == 0)
189  precision = 32767;
190  for (p = s; *p && (--precision >= 0); p++)
191  ;
192  break;
193  case 'D':
194  case 'd':
195  case 'I':
196  case 'i':
197  if (is_long)
198  l = va_arg(ap, long);
199  else
200  l = va_arg(ap, int);
201  if (l < 0) {
202  *p++ = '-';
203  l = -l;
204  }
205  p = ltoa(p, l, 10);
206  break;
207 #if CHPRINTF_USE_FLOAT
208  case 'f':
209  f = (float) va_arg(ap, double);
210  if (f < 0) {
211  *p++ = '-';
212  f = -f;
213  }
214  p = ftoa(p, f);
215  break;
216 #endif
217  case 'X':
218  case 'x':
219  c = 16;
220  goto unsigned_common;
221  case 'U':
222  case 'u':
223  c = 10;
224  goto unsigned_common;
225  case 'O':
226  case 'o':
227  c = 8;
228 unsigned_common:
229  if (is_long)
230  l = va_arg(ap, unsigned long);
231  else
232  l = va_arg(ap, unsigned int);
233  p = ltoa(p, l, c);
234  break;
235  default:
236  *p++ = c;
237  break;
238  }
239  i = (int)(p - s);
240  if ((width -= i) < 0)
241  width = 0;
242  if (left_align == FALSE)
243  width = -width;
244  if (width < 0) {
245  if (*s == '-' && filler == '0') {
246  chSequentialStreamPut(chp, (uint8_t)*s++);
247  i--;
248  }
249  do {
250  chSequentialStreamPut(chp, (uint8_t)filler);
251  } while (++width != 0);
252  }
253  while (--i >= 0)
254  chSequentialStreamPut(chp, (uint8_t)*s++);
255 
256  while (width) {
257  chSequentialStreamPut(chp, (uint8_t)filler);
258  width--;
259  }
260  }
261 }

267  * The general parameters format is: %[-][width|*][.precision|*][l|L]p.
268  * The following parameter types (p) are supported:
269  * - <b>x</b> hexadecimal integer.
270  * - <b>X</b> hexadecimal long.
271  * - <b>o</b> octal integer.
272  * - <b>O</b> octal long.
273  * - <b>d</b> decimal signed integer.
274  * - <b>D</b> decimal signed long.
275  * - <b>u</b> decimal unsigned integer.
276  * - <b>U</b> decimal unsigned long.
277  * - <b>c</b> character.
278  * - <b>s</b> string.
279  * .
280  *
281  * @param[in] str pointer to a buffer
282  * @param[in] size maximum size of the buffer
283  * @param[in] fmt formatting string
284  * @return The size of the generated string.
285  *
286  * @api
287  */
288 int chsnprintf(char *str, size_t size, const char *fmt, ...) {
289  va_list ap;
290  MemoryStream ms;
291  BaseSequentialStream *chp;
292 
293  /* Memory stream object to be used as a string writer, reserving one
294  byte for the final zero.*/
295  msObjectInit(&ms, (uint8_t *)str, size - 1, 0);
296 
297  /* Performing the print operation using the common code.*/
298  chp = (BaseSequentialStream *)&ms;
299  va_start(ap, fmt);
300  chvprintf(chp, fmt, ap);
301  va_end(ap);
302 
303  /* Final zero and size return.*/
304  str[ms.eos] = 0;
305  return ms.eos;
306 }
