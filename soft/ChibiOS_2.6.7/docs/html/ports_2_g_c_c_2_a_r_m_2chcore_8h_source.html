217  tp->p_ctx.r13->r4 = pf; \

434 /**
435  * @brief Performs a context switch between two threads.
436  * @details This is the most critical code in any port, this function
437  * is responsible for the context switch between 2 threads.
438  * @note The implementation of this code affects <b>directly</b> the context
439  * switch performance so optimize here as much as you can.
440  * @note Implemented as inlined code for performance reasons.
441  *
442  * @param[in] ntp the thread to be switched in
443  * @param[in] otp the thread to be switched out
444  */
445 #ifdef THUMB
446 #if CH_DBG_ENABLE_STACK_CHECK
447 #define port_switch(ntp, otp) { \
448  register struct intctx *r13 asm ("r13"); \
449  if ((stkalign_t *)(r13 - 1) < otp->p_stklimit) \
450  chDbgPanic("stack overflow"); \
451  _port_switch_thumb(ntp, otp); \
452 }
453 #else /* !CH_DBG_ENABLE_STACK_CHECK */
454 #define port_switch(ntp, otp) _port_switch_thumb(ntp, otp)
455 #endif /* !CH_DBG_ENABLE_STACK_CHECK */
456 #else /* !THUMB */
457 #if CH_DBG_ENABLE_STACK_CHECK
458 #define port_switch(ntp, otp) { \
459  register struct intctx *r13 asm ("r13"); \
460  if ((stkalign_t *)(r13 - 1) < otp->p_stklimit) \
461  chDbgPanic("stack overflow"); \
462  _port_switch_arm(ntp, otp); \
463 }
464 #else /* !CH_DBG_ENABLE_STACK_CHECK */
465 #define port_switch(ntp, otp) _port_switch_arm(ntp, otp)
466 #endif /* !CH_DBG_ENABLE_STACK_CHECK */
467 #endif /* !THUMB */
