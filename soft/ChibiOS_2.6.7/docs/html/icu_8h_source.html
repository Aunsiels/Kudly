144  *
145  * @param[in] icup pointer to the @p ICUDriver object
146  *
147  * @notapi
148  */
149 #define _icu_isr_invoke_width_cb(icup) { \
150  if ((icup)->state != ICU_WAITING) { \
151  (icup)->state = ICU_IDLE; \
152  (icup)->config->width_cb(icup); \
153  } \
154 }

156 /**
157  * @brief Common ISR code, ICU period event.
158  *
159  * @param[in] icup pointer to the @p ICUDriver object
160  *
161  * @notapi
162  */
163 #define _icu_isr_invoke_period_cb(icup) { \
164  icustate_t previous_state = (icup)->state; \
165  (icup)->state = ICU_ACTIVE; \
166  if (previous_state != ICU_WAITING) \
167  (icup)->config->period_cb(icup); \
168 }
