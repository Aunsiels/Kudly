49 /*
50  * Performs a context switch between two threads.
51  */
52  EXPORT _port_switch
53 _port_switch PROC
54  push {r4, r5, r6, r7, lr}
55  mov r4, r8
56  mov r5, r9
57  mov r6, r10
58  mov r7, r11
59  push {r4, r5, r6, r7}
60  mov r3, sp
61  str r3, [r1, #CONTEXT_OFFSET]
62  ldr r3, [r0, #CONTEXT_OFFSET]
63  mov sp, r3
64  pop {r4, r5, r6, r7}
65  mov r8, r4
66  mov r9, r5
67  mov r10, r6
68  mov r11, r7
69  pop {r4, r5, r6, r7, pc}
70  ENDP

72 /*
73  * Start a thread by invoking its work function.
74  * If the work function returns @p chThdExit() is automatically invoked.
75  */
76  EXPORT _port_thread_start
77 _port_thread_start PROC
78 #if CH_DBG_SYSTEM_STATE_CHECK
79  bl dbg_check_unlock
80 #endif
81  cpsie i
82  mov r0, r5
83  blx r4
84  bl chThdExit
85  ENDP

87 /*
88  * Post-IRQ switch code.
89  * Exception handlers return here for context switching.
90  */
91  EXPORT _port_switch_from_isr
92  EXPORT _port_exit_from_isr
93 _port_switch_from_isr PROC
94 #if CH_DBG_SYSTEM_STATE_CHECK
95  bl dbg_check_lock
96 #endif
97  bl chSchDoReschedule
98 #if CH_DBG_SYSTEM_STATE_CHECK
99  bl dbg_check_unlock
100 #endif
101 _port_exit_from_isr
102  ldr r2, =SCB_ICSR
103  movs r3, #128
104 #if CORTEX_ALTERNATE_SWITCH
105  lsls r3, r3, #21
106  str r3, [r2, #0]
107  cpsie i
108 #else
109  lsls r3, r3, #24
110  str r3, [r2, #0]
111 #endif
112 waithere b waithere
113  ENDP
