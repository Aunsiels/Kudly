489 /**
490  * @brief Performs a context switch between two threads.
491  * @details This is the most critical code in any port, this function
492  * is responsible for the context switch between 2 threads.
493  * @note The implementation of this code affects <b>directly</b> the context
494  * switch performance so optimize here as much as you can.
495  *
496  * @param[in] ntp the thread to be switched in
497  * @param[in] otp the thread to be switched out
498  */
499 #if !CH_DBG_ENABLE_STACK_CHECK || defined(__DOXYGEN__)
500 #define port_switch(ntp, otp) _port_switch(ntp, otp)
501 #else
502 #define port_switch(ntp, otp) { \
503  register struct intctx *r13 asm ("r13"); \
504  if ((stkalign_t *)(r13 - 1) < otp->p_stklimit) \
505  chDbgPanic("stack overflow"); \
506  _port_switch(ntp, otp); \
507 }
