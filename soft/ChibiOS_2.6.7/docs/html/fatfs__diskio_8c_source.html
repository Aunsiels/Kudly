46 
47  switch (drv) {

110 
111 DRESULT disk_read (
112  BYTE drv, /* Physical drive nmuber (0..) */
113  BYTE *buff, /* Data buffer to store read data */
114  DWORD sector, /* Sector address (LBA) */
115  BYTE count /* Number of sectors to read (1..255) */
116 )
117 {
118  switch (drv) {
119 #if HAL_USE_MMC_SPI
120  case MMC:
121  if (blkGetDriverState(&MMCD1) != BLK_READY)
122  return RES_NOTRDY;
123  if (mmcStartSequentialRead(&MMCD1, sector))
124  return RES_ERROR;
125  while (count > 0) {
126  if (mmcSequentialRead(&MMCD1, buff))
127  return RES_ERROR;
128  buff += MMCSD_BLOCK_SIZE;
129  count--;
130  }
131  if (mmcStopSequentialRead(&MMCD1))
132  return RES_ERROR;
133  return RES_OK;
134 #else
135  case SDC:
136  if (blkGetDriverState(&SDCD1) != BLK_READY)
137  return RES_NOTRDY;
138  if (sdcRead(&SDCD1, sector, buff, count))
139  return RES_ERROR;
140  return RES_OK;
141 #endif
142  }
143  return RES_PARERR;
144 }

195 DRESULT disk_ioctl (
196  BYTE drv, /* Physical drive nmuber (0..) */
197  BYTE ctrl, /* Control code */
198  void *buff /* Buffer to send/receive control data */
199 )
200 {
201  switch (drv) {
202 #if HAL_USE_MMC_SPI
203  case MMC:
204  switch (ctrl) {
205  case CTRL_SYNC:
206  return RES_OK;
207  case GET_SECTOR_SIZE:
208  *((WORD *)buff) = MMCSD_BLOCK_SIZE;
209  return RES_OK;
210 #if _USE_ERASE
211  case CTRL_ERASE_SECTOR:
212  mmcErase(&MMCD1, *((DWORD *)buff), *((DWORD *)buff + 1));
213  return RES_OK;
214 #endif
215  default:
216  return RES_PARERR;
217  }
218 #else
219  case SDC:
220  switch (ctrl) {
221  case CTRL_SYNC:
222  return RES_OK;
223  case GET_SECTOR_COUNT:
224  *((DWORD *)buff) = mmcsdGetCardCapacity(&SDCD1);
225  return RES_OK;
226  case GET_SECTOR_SIZE:
227  *((WORD *)buff) = MMCSD_BLOCK_SIZE;
228  return RES_OK;
229  case GET_BLOCK_SIZE:
230  *((DWORD *)buff) = 256; /* 512b blocks in one erase block */
231  return RES_OK;
232 #if _USE_ERASE
233  case CTRL_ERASE_SECTOR:
234  sdcErase(&SDCD1, *((DWORD *)buff), *((DWORD *)buff + 1));
235  return RES_OK;
236 #endif
237  default:
238  return RES_PARERR;
239  }
240 #endif
241  }
242  return RES_PARERR;
243 }
