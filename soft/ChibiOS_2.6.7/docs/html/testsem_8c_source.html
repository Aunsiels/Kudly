145 static void sem2_execute(void) {
146  int i;
147  systime_t target_time;
148  msg_t msg;
149 
150  /*
151  * Testing special case TIME_IMMEDIATE.
152  */
153  msg = chSemWaitTimeout(&sem1, TIME_IMMEDIATE);
154  test_assert(1, msg == RDY_TIMEOUT, "wrong wake-up message");
155  test_assert(2, isempty(&sem1.s_queue), "queue not empty");
156  test_assert(3, sem1.s_cnt == 0, "counter not zero");
157 
158  /*
159  * Testing not timeout condition.
160  */
161  threads[0] = chThdCreateStatic(wa[0], WA_SIZE, chThdGetPriority() - 1,
162  thread2, 0);
163  msg = chSemWaitTimeout(&sem1, MS2ST(500));
164  test_wait_threads();
165  test_assert(4, msg == RDY_OK, "wrong wake-up message");
166  test_assert(5, isempty(&sem1.s_queue), "queue not empty");
167  test_assert(6, sem1.s_cnt == 0, "counter not zero");
168 
169  /*
170  * Testing timeout condition.
171  */
172  test_wait_tick();
173  target_time = chTimeNow() + MS2ST(5 * 500);
174  for (i = 0; i < 5; i++) {
175  test_emit_token('A' + i);
176  msg = chSemWaitTimeout(&sem1, MS2ST(500));
177  test_assert(7, msg == RDY_TIMEOUT, "wrong wake-up message");
178  test_assert(8, isempty(&sem1.s_queue), "queue not empty");
179  test_assert(9, sem1.s_cnt == 0, "counter not zero");
180  }
181  test_assert_sequence(10, "ABCDE");
182  test_assert_time_window(11, target_time, target_time + ALLOWED_DELAY);
183 }

252 static void sem4_execute(void) {
253  BinarySemaphore bsem;
254 
255  /* Creates a taken binary semaphore.*/
256  chBSemInit(&bsem, TRUE);
257  chBSemReset(&bsem, TRUE);
258  test_assert(1, chBSemGetStateI(&bsem) == TRUE, "not taken");
259 
260  /* Starts a signaler thread at a lower priority.*/
261  threads[0] = chThdCreateStatic(wa[0], WA_SIZE,
262  chThdGetPriority()-1, thread4, &bsem);
263 
264  /* Waits to be signaled.*/
265  chBSemWait(&bsem);
266 
267  /* The binary semaphore is expected to be taken.*/
268  test_assert(2, chBSemGetStateI(&bsem) == TRUE, "not taken");
269 
270  /* Releasing it, check both the binary semaphore state and the underlying
271  counter semaphore state..*/
272  chBSemSignal(&bsem);
273  test_assert(3, chBSemGetStateI(&bsem) == FALSE, "still taken");
274  test_assert(4, chSemGetCounterI(&bsem.bs_sem) == 1, "unexpected counter");
275 
276  /* Checking signaling overflow, the counter must not go beyond 1.*/
277  chBSemSignal(&bsem);
278  test_assert(3, chBSemGetStateI(&bsem) == FALSE, "taken");
279  test_assert(5, chSemGetCounterI(&bsem.bs_sem) == 1, "unexpected counter");
280 }
