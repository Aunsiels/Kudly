71 static void pools1_execute(void) {
72  int i;
73 
74  /* Adding the WAs to the pool.*/
75  chPoolLoadArray(&mp1, wa[0], MAX_THREADS);
76 
77  /* Emptying the pool.*/
78  for (i = 0; i < MAX_THREADS; i++)
79  test_assert(1, chPoolAlloc(&mp1) != NULL, "list empty");
80 
81  /* Now must be empty.*/
82  test_assert(2, chPoolAlloc(&mp1) == NULL, "list not empty");
83 
84  /* Adding the WAs to the pool, one by one this time.*/
85  for (i = 0; i < MAX_THREADS; i++)
86  chPoolFree(&mp1, wa[i]);
87 
88  /* Emptying the pool again.*/
89  for (i = 0; i < MAX_THREADS; i++)
90  test_assert(3, chPoolAlloc(&mp1) != NULL, "list empty");
91 
92  /* Now must be empty again.*/
93  test_assert(4, chPoolAlloc(&mp1) == NULL, "list not empty");
94 
95  /* Covering the case where a provider is unable to return more memory.*/
96  chPoolInit(&mp1, 16, null_provider);
97  test_assert(5, chPoolAlloc(&mp1) == NULL, "provider returned memory");
98 }
