46 /**
47  * @brief System formatted output function.
48  * @details This function implements a minimal @p printf() like functionality
49  * with output on a @p BaseSequentialStream.
50  * The general parameters format is: %[-][width|*][.precision|*][l|L]p.
51  * The following parameter types (p) are supported:
52  * - <b>x</b> hexadecimal integer.
53  * - <b>X</b> hexadecimal long.
54  * - <b>o</b> octal integer.
55  * - <b>O</b> octal long.
56  * - <b>d</b> decimal signed integer.
57  * - <b>D</b> decimal signed long.
58  * - <b>u</b> decimal unsigned integer.
59  * - <b>U</b> decimal unsigned long.
60  * - <b>c</b> character.
61  * - <b>s</b> string.
62  * .
63  *
64  * @param[in] chp pointer to a @p BaseSequentialStream implementing object
65  * @param[in] fmt formatting string
66  *
67  * @api
68  */
69 static INLINE void chprintf(BaseSequentialStream *chp, const char *fmt, ...) {
70  va_list ap;
71 
72  va_start(ap, fmt);
73  chvprintf(chp, fmt, ap);
74  va_end(ap);
75 }
