59 /**
60  * @brief SVC vector.
61  * @details The SVC vector is used for exception mode re-entering after a
62  * context switch.
63  * @note The PendSV vector is only used in advanced kernel mode.
64  */
65 void SVCallVector(void) {
66  struct extctx *ctxp;
67  register uint32_t psp __asm("psp");
68 
69 #if CORTEX_USE_FPU
70  /* Enforcing unstacking of the FP part of the context.*/
71  SCB_FPCCR &= ~FPCCR_LSPACT;
72 #endif
73 
74  /* Current PSP value.*/
75  ctxp = (struct extctx *)psp;
76 
77  /* Discarding the current exception context and positioning the stack to
78  point to the real one.*/
79  ctxp++;
80 
81  /* Restoring real position of the original stack frame.*/
82  psp = (uint32_t)ctxp;
83  port_unlock_from_isr();
84 }
85 #endif /* !CORTEX_SIMPLIFIED_PRIORITY */

119 /**
120  * @brief Port-related initialization code.
121  */
122 void _port_init(void) {
123 
124  /* Initialization of the vector table and priority related settings.*/
125  SCB_VTOR = CORTEX_VTOR_INIT;
126  SCB_AIRCR = AIRCR_VECTKEY | AIRCR_PRIGROUP(CORTEX_PRIGROUP_INIT);
127 
128 #if CORTEX_USE_FPU
129  {
130  register uint32_t control __asm("control");
131  register uint32_t fpscr __asm("fpscr");
132 
133  /* Initializing the FPU context save in lazy mode.*/
134  SCB_FPCCR = FPCCR_ASPEN | FPCCR_LSPEN;
135 
136  /* CP10 and CP11 set to full access in the startup code.*/
137 /* SCB_CPACR |= 0x00F00000;*/
138 
139  /* Enables FPU context save/restore on exception entry/exit (FPCA bit).*/
140  control |= 4;
141 
142  /* FPSCR and FPDSCR initially zero.*/
143  fpscr = 0;
144  SCB_FPDSCR = 0;
145  }
146 #endif
147 
148  /* Initialization of the system vectors used by the port.*/
149  nvicSetSystemHandlerPriority(HANDLER_SVCALL,
150  CORTEX_PRIORITY_MASK(CORTEX_PRIORITY_SVCALL));
151  nvicSetSystemHandlerPriority(HANDLER_PENDSV,
152  CORTEX_PRIORITY_MASK(CORTEX_PRIORITY_PENDSV));
153  nvicSetSystemHandlerPriority(HANDLER_SYSTICK,
154  CORTEX_PRIORITY_MASK(CORTEX_PRIORITY_SYSTICK));
155 }

157 /**
158  * @brief Exception exit redirection to _port_switch_from_isr().
159  */
160 void _port_irq_epilogue(void) {
161 
162  port_lock_from_isr();
163  if ((SCB_ICSR & ICSR_RETTOBASE) != 0) {
164  struct extctx *ctxp;
165  register uint32_t psp __asm("psp");
166 
167  /* Current PSP value.*/
168  ctxp = (struct extctx *)psp;
169 
170 #if CORTEX_USE_FPU
171  /* Enforcing a lazy FPU state save. Note, it goes in the original
172  context because the FPCAR register has not been modified.*/
173  {
174  volatile register uint32_t fpscr __asm("fpscr");
175  (void)fpscr;
176  }
177 #endif
178 
179  /* Adding an artificial exception return context, there is no need to
180  populate it fully.*/
181  ctxp--;
182  ctxp->xpsr = (regarm_t)0x01000000;
183 #if CORTEX_USE_FPU
184  ctxp->fpscr = (regarm_t)SCB_FPDSCR;
185 #endif
186  psp = (uint32_t)ctxp;
187 
188  /* The exit sequence is different depending on if a preemption is
189  required or not.*/
190  if (chSchIsPreemptionRequired()) {
191  /* Preemption is required we need to enforce a context switch.*/
192  ctxp->pc = (regarm_t)_port_switch_from_isr;
193  }
194  else {
195  /* Preemption not required, we just need to exit the exception
196  atomically.*/
197  ctxp->pc = (regarm_t)_port_exit_from_isr;
198  }
199 
200  /* Note, returning without unlocking is intentional, this is done in
201  order to keep the rest of the context switch atomic.*/
202  return;
203  }
204  port_unlock_from_isr();
205 }
