349 /**
350  * @brief Performs a context switch between two threads.
351  * @details This is the most critical code in any port, this function
352  * is responsible for the context switch between 2 threads.
353  * @note The implementation of this code affects <b>directly</b> the context
354  * switch performance so optimize here as much as you can.
355  *
356  * @param[in] ntp the thread to be switched in
357  * @param[in] otp the thread to be switched out
358  */
359 #if !CH_DBG_ENABLE_STACK_CHECK || defined(__DOXYGEN__)
360 #define port_switch(ntp, otp) _port_switch(ntp, otp)
361 #else
362 #define port_switch(ntp, otp) { \
363  uint8_t *r13 = (uint8_t *)__current_sp(); \
364  if ((stkalign_t *)(r13 - sizeof(struct intctx)) < otp->p_stklimit) \
365  chDbgPanic("stack overflow"); \
366  _port_switch(ntp, otp); \
367 }
368 #endif
