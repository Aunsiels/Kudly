83 static void dyn1_execute(void) {
84  size_t n, sz;
85  void *p1;
86  tprio_t prio = chThdGetPriority();
87 
88  (void)chHeapStatus(&heap1, &sz);
89  /* Starting threads from the heap. */
90  threads[0] = chThdCreateFromHeap(&heap1, THD_WA_SIZE(THREADS_STACK_SIZE),
91  prio-1, thread, "A");
92  threads[1] = chThdCreateFromHeap(&heap1, THD_WA_SIZE(THREADS_STACK_SIZE),
93  prio-2, thread, "B");
94  /* Allocating the whole heap in order to make the thread creation fail.*/
95  (void)chHeapStatus(&heap1, &n);
96  p1 = chHeapAlloc(&heap1, n);
97  threads[2] = chThdCreateFromHeap(&heap1, THD_WA_SIZE(THREADS_STACK_SIZE),
98  prio-3, thread, "C");
99  chHeapFree(p1);
100 
101  test_assert(1, (threads[0] != NULL) &&
102  (threads[1] != NULL) &&
103  (threads[2] == NULL) &&
104  (threads[3] == NULL) &&
105  (threads[4] == NULL),
106  "thread creation failed");
107 
108  /* Claiming the memory from terminated threads. */
109  test_wait_threads();
110  test_assert_sequence(2, "AB");
111 
112  /* Heap status checked again.*/
113  test_assert(3, chHeapStatus(&heap1, &n) == 1, "heap fragmented");
114  test_assert(4, n == sz, "heap size changed");
115 }

191 static bool_t regfind(Thread *tp) {
192  Thread *ftp;

194

196  do {

208 static void dyn3_execute(void) {
209  Thread *tp;
210  tprio_t prio = chThdGetPriority();
211 
212  /* Testing references increase/decrease and final detach.*/
213  tp = chThdCreateFromHeap(&heap1, WA_SIZE, prio-1, thread, "A");
214  test_assert(1, tp->p_refs == 1, "wrong initial reference counter");
215  chThdAddRef(tp);
216  test_assert(2, tp->p_refs == 2, "references increase failure");
217  chThdRelease(tp);
218  test_assert(3, tp->p_refs == 1, "references decrease failure");
219 
220  /* Verify the new threads count.*/
221  test_assert(4, regfind(tp), "thread missing from registry");
222  test_assert(5, regfind(tp), "thread disappeared");
223 
224  /* Detach and let the thread execute and terminate.*/
225  chThdRelease(tp);
226  test_assert(6, tp->p_refs == 0, "detach failure");
227  test_assert(7, tp->p_state == THD_STATE_READY, "invalid state");
228  test_assert(8, regfind(tp), "thread disappeared");
229  test_assert(9, regfind(tp), "thread disappeared");
230  chThdSleepMilliseconds(50); /* The thread just terminates. */
231  test_assert(10, tp->p_state == THD_STATE_FINAL, "invalid state");
232 
233  /* Clearing the zombie by scanning the registry.*/
234  test_assert(11, regfind(tp), "thread disappeared");
235  test_assert(12, !regfind(tp), "thread still in registry");
236 }
