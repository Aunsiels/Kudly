117  * @page test_threads_003 Threads priority change test
118  *
119  * <h2>Description</h2>
120  * A series of priority changes are performed on the current thread in order
121  * to verify that the priority change happens as expected.<br>
122  * If the @p CH_USE_MUTEXES option is enabled then the priority changes are
123  * also tested under priority inheritance boosted priority state.
124  */
125 
126 static void thd3_execute(void) {
127  tprio_t prio, p1;
128 
129  prio = chThdGetPriority();
130  p1 = chThdSetPriority(prio + 1);
131  test_assert(1, p1 == prio,
132  "unexpected returned priority level");
133  test_assert(2, chThdGetPriority() == prio + 1,
134  "unexpected priority level");
135  p1 = chThdSetPriority(p1);
136  test_assert(3, p1 == prio + 1,
137  "unexpected returned priority level");
138  test_assert(4, chThdGetPriority() == prio,
139  "unexpected priority level");
140 
141 #if CH_USE_MUTEXES || defined(__DOXYGEN__)
142  /* Simulates a priority boost situation (p_prio > p_realprio).*/
143  chSysLock();
144  chThdSelf()->p_prio += 2;
145  chSysUnlock();
146  test_assert(5, chThdGetPriority() == prio + 2,
147  "unexpected priority level");
148 
149  /* Tries to raise but below the boost level. */
150  p1 = chThdSetPriority(prio + 1);
151  test_assert(6, p1 == prio,
152  "unexpected returned priority level");
153  test_assert(7, chThdSelf()->p_prio == prio + 2,
154  "unexpected priority level");
155  test_assert(8, chThdSelf()->p_realprio == prio + 1,
156  "unexpected returned real priority level");
157 
158  /* Tries to raise above the boost level. */
159  p1 = chThdSetPriority(prio + 3);
160  test_assert(9, p1 == prio + 1,
161  "unexpected returned priority level");
162  test_assert(10, chThdSelf()->p_prio == prio + 3,
163  "unexpected priority level");
164  test_assert(11, chThdSelf()->p_realprio == prio + 3,
165  "unexpected real priority level");
166 
167  chSysLock();
168  chThdSelf()->p_prio = prio;
169  chThdSelf()->p_realprio = prio;
170  chSysUnlock();
171 #endif
172 }

189 static void thd4_execute(void) {
190  systime_t time;
191 
192  test_wait_tick();
193 
194  /* Timeouts in microseconds.*/
195  time = chTimeNow();
196  chThdSleepMicroseconds(100000);
197  test_assert_time_window(1, time + US2ST(100000), time + US2ST(100000) + 1);
198 
199  /* Timeouts in milliseconds.*/
200  time = chTimeNow();
201  chThdSleepMilliseconds(100);
202  test_assert_time_window(2, time + MS2ST(100), time + MS2ST(100) + 1);
203 
204  /* Timeouts in seconds.*/
205  time = chTimeNow();
206  chThdSleepSeconds(1);
207  test_assert_time_window(3, time + S2ST(1), time + S2ST(1) + 1);
208 
209  /* Absolute timelines.*/
210  time = chTimeNow() + MS2ST(100);
211  chThdSleepUntil(time);
212  test_assert_time_window(4, time, time + 1);
213 }
