39 /*
40  * Generic MII registers. Note, not all registers are present on all PHY
41  * devices and some extra registers may be present.
42  */
43 #define MII_BMCR 0x00 /**< Basic mode control register. */
44 #define MII_BMSR 0x01 /**< Basic mode status register. */
45 #define MII_PHYSID1 0x02 /**< PHYS ID 1. */
46 #define MII_PHYSID2 0x03 /**< PHYS ID 2. */
47 #define MII_ADVERTISE 0x04 /**< Advertisement control reg. */
48 #define MII_LPA 0x05 /**< Link partner ability reg. */
49 #define MII_EXPANSION 0x06 /**< Expansion register. */
50 #define MII_ANNPTR 0x07 /**< 1000BASE-T control. */
51 #define MII_CTRL1000 0x09 /**< 1000BASE-T control. */
52 #define MII_STAT1000 0x0a /**< 1000BASE-T status. */
53 #define MII_ESTATUS 0x0f /**< Extended Status. */
54 #define MII_PHYSTS 0x10 /**< PHY Status register. */
55 #define MII_MICR 0x11 /**< MII Interrupt ctrl register. */
56 #define MII_DCOUNTER 0x12 /**< Disconnect counter. */
57 #define MII_FCSCOUNTER 0x13 /**< False carrier counter. */
58 #define MII_NWAYTEST 0x14 /**< N-way auto-neg test reg. */
59 #define MII_RERRCOUNTER 0x15 /**< Receive error counter. */
60 #define MII_SREVISION 0x16 /**< Silicon revision. */
61 #define MII_RESV1 0x17 /**< Reserved. */
62 #define MII_LBRERROR 0x18 /**< Lpback, rx, bypass error. */
63 #define MII_PHYADDR 0x19 /**< PHY address. */
64 #define MII_RESV2 0x1a /**< Reserved. */
65 #define MII_TPISTATUS 0x1b /**< TPI status for 10Mbps. */
66 #define MII_NCONFIG 0x1c /**< Network interface config. */

68 /*
69  * Basic mode control register.
70  */
71 #define BMCR_RESV 0x007f /**< Unused. */
72 #define BMCR_CTST 0x0080 /**< Collision test. */
73 #define BMCR_FULLDPLX 0x0100 /**< Full duplex. */
74 #define BMCR_ANRESTART 0x0200 /**< Auto negotiation restart. */
75 #define BMCR_ISOLATE 0x0400 /**< Disconnect DP83840 from MII. */
76 #define BMCR_PDOWN 0x0800 /**< Powerdown. */
77 #define BMCR_ANENABLE 0x1000 /**< Enable auto negotiation. */
78 #define BMCR_SPEED100 0x2000 /**< Select 100Mbps. */
79 #define BMCR_LOOPBACK 0x4000 /**< TXD loopback bit. */
80 #define BMCR_RESET 0x8000 /**< Reset. */
81 
82 /*
83  * Basic mode status register.
84  */
85 #define BMSR_ERCAP 0x0001 /**< Ext-reg capability. */
86 #define BMSR_JCD 0x0002 /**< Jabber detected. */
87 #define BMSR_LSTATUS 0x0004 /**< Link status. */
88 #define BMSR_ANEGCAPABLE 0x0008 /**< Able to do auto-negotiation. */
89 #define BMSR_RFAULT 0x0010 /**< Remote fault detected. */
90 #define BMSR_ANEGCOMPLETE 0x0020 /**< Auto-negotiation complete. */
91 #define BMSR_MFPRESUPPCAP 0x0040 /**< Able to suppress preamble. */
92 #define BMSR_RESV 0x0780 /**< Unused. */
93 #define BMSR_10HALF 0x0800 /**< Can do 10mbps, half-duplex. */
94 #define BMSR_10FULL 0x1000 /**< Can do 10mbps, full-duplex. */
95 #define BMSR_100HALF 0x2000 /**< Can do 100mbps, half-duplex. */
96 #define BMSR_100FULL 0x4000 /**< Can do 100mbps, full-duplex. */
97 #define BMSR_100BASE4 0x8000 /**< Can do 100mbps, 4k packets. */
98 
99 /*
100  * Advertisement control register.
101  */
102 #define ADVERTISE_SLCT 0x001f /**< Selector bits. */
103 #define ADVERTISE_CSMA 0x0001 /**< Only selector supported. */
104 #define ADVERTISE_10HALF 0x0020 /**< Try for 10mbps half-duplex. */
105 #define ADVERTISE_10FULL 0x0040 /**< Try for 10mbps full-duplex. */
106 #define ADVERTISE_100HALF 0x0080 /**< Try for 100mbps half-duplex. */
107 #define ADVERTISE_100FULL 0x0100 /**< Try for 100mbps full-duplex. */
108 #define ADVERTISE_100BASE4 0x0200 /**< Try for 100mbps 4k packets. */
109 #define ADVERTISE_PAUSE_CAP 0x0400 /**< Try for pause. */
110 #define ADVERTISE_PAUSE_ASYM 0x0800 /**< Try for asymetric pause. */
111 #define ADVERTISE_RESV 0x1000 /**< Unused. */
112 #define ADVERTISE_RFAULT 0x2000 /**< Say we can detect faults. */
113 #define ADVERTISE_LPACK 0x4000 /**< Ack link partners response. */
114 #define ADVERTISE_NPAGE 0x8000 /**< Next page bit. */
115 
116 #define ADVERTISE_FULL (ADVERTISE_100FULL | ADVERTISE_10FULL | \
117  ADVERTISE_CSMA)
118 #define ADVERTISE_ALL (ADVERTISE_10HALF | ADVERTISE_10FULL | \
119  ADVERTISE_100HALF | ADVERTISE_100FULL)
120 
121 /*
122  * Link partner ability register.
123  */
124 #define LPA_SLCT 0x001f /**< Same as advertise selector. */
125 #define LPA_10HALF 0x0020 /**< Can do 10mbps half-duplex. */
126 #define LPA_10FULL 0x0040 /**< Can do 10mbps full-duplex. */
127 #define LPA_100HALF 0x0080 /**< Can do 100mbps half-duplex. */
128 #define LPA_100FULL 0x0100 /**< Can do 100mbps full-duplex. */
129 #define LPA_100BASE4 0x0200 /**< Can do 100mbps 4k packets. */
130 #define LPA_PAUSE_CAP 0x0400 /**< Can pause. */
131 #define LPA_PAUSE_ASYM 0x0800 /**< Can pause asymetrically. */
132 #define LPA_RESV 0x1000 /**< Unused. */
133 #define LPA_RFAULT 0x2000 /**< Link partner faulted. */
134 #define LPA_LPACK 0x4000 /**< Link partner acked us. */
135 #define LPA_NPAGE 0x8000 /**< Next page bit. */
136 
137 #define LPA_DUPLEX (LPA_10FULL | LPA_100FULL)
138 #define LPA_100 (LPA_100FULL | LPA_100HALF | LPA_100BASE4)
