42 void port_init(void) {
43 #if PPC_SUPPORTS_IVORS

47  "mtIVOR4 %%r3 \t\n"
48  "li %%r3, _IVOR10@l \t\n"
49  "mtIVOR10 %%r3" : : : "memory");
50 #endif
51 }

67 /**
68  * @brief Performs a context switch between two threads.
69  * @details This is the most critical code in any port, this function
70  * is responsible for the context switch between 2 threads.
71  * @note The implementation of this code affects <b>directly</b> the context
72  * switch performance so optimize here as much as you can.
73  */
74 #if !defined(__DOXYGEN__)
75 __attribute__((naked))
76 #endif
77 void port_dummy1(void) {
78 
79  asm (".global _port_switch");
80  asm ("_port_switch:");
81  asm ("subi %sp, %sp, 80"); /* Size of the intctx structure. */
82  asm ("mflr %r0");
83  asm ("stw %r0, 84(%sp)"); /* LR into the caller frame. */
84  asm ("mfcr %r0");
85  asm ("stw %r0, 0(%sp)"); /* CR. */
86  asm ("stmw %r14, 4(%sp)"); /* GPR14...GPR31. */
87 
88  asm ("stw %sp, 12(%r4)"); /* Store swapped-out stack. */
89  asm ("lwz %sp, 12(%r3)"); /* Load swapped-in stack. */
90 
91  asm ("lmw %r14, 4(%sp)"); /* GPR14...GPR31. */
92  asm ("lwz %r0, 0(%sp)"); /* CR. */
93  asm ("mtcr %r0");
94  asm ("lwz %r0, 84(%sp)"); /* LR from the caller frame. */
95  asm ("mtlr %r0");
96  asm ("addi %sp, %sp, 80"); /* Size of the intctx structure. */
97  asm ("blr");
98 }
