101 u32_t sys_arch_sem_wait(sys_sem_t *sem, u32_t timeout) {
102  systime_t time, tmo;
103 
104  chSysLock();
105  tmo = timeout > 0 ? (systime_t)timeout : TIME_INFINITE;
106  time = chTimeNow();
107  if (chSemWaitTimeoutS(*sem, tmo) != RDY_OK)
108  time = SYS_ARCH_TIMEOUT;
109  else
110  time = chTimeNow() - time;
111  chSysUnlock();
112  return time;
113 }

125 err_t sys_mbox_new(sys_mbox_t *mbox, int size) {

128  if (*mbox == 0) {

139 void sys_mbox_free(sys_mbox_t *mbox) {
140 
141  if (chMBGetUsedCountI(*mbox) != 0) {
142  // If there are messages still present in the mailbox when the mailbox
143  // is deallocated, it is an indication of a programming error in lwIP
144  // and the developer should be notified.
145  SYS_STATS_INC(mbox.err);
146  chMBReset(*mbox);
147  }
148  chHeapFree(*mbox);
149  *mbox = SYS_MBOX_NULL;
150  SYS_STATS_DEC(mbox.used);
151 }

167 u32_t sys_arch_mbox_fetch(sys_mbox_t *mbox, void **msg, u32_t timeout) {
168  systime_t time, tmo;
169 
170  chSysLock();
171  tmo = timeout > 0 ? (systime_t)timeout : TIME_INFINITE;
172  time = chTimeNow();
173  if (chMBFetchS(*mbox, (msg_t *)msg, tmo) != RDY_OK)
174  time = SYS_ARCH_TIMEOUT;

176  time = chTimeNow() - time;
177  chSysUnlock();

224 u32_t sys_now(void) {
225 
226 #if CH_FREQUENCY == 1000
227  return (u32_t)chTimeNow();
228 #elif (CH_FREQUENCY / 1000) >= 1 && (CH_FREQUENCY % 1000) == 0
229  return ((u32_t)chTimeNow() - 1) / (CH_FREQUENCY / 1000) + 1;
230 #elif (1000 / CH_FREQUENCY) >= 1 && (1000 % CH_FREQUENCY) == 0
231  return ((u32_t)chTimeNow() - 1) * (1000 / CH_FREQUENCY) + 1;
232 #else
233  return (u32_t)(((u64_t)(chTimeNow() - 1) * 1000) / CH_FREQUENCY) + 1;
234 #endif
235 }
