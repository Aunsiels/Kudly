86 /**
87  * @brief Prints a decimal unsigned number.
88  *
89  * @param[in] n the number to be printed
90  */
91 void test_printn(uint32_t n) {
92  char buf[16], *p;
93 
94  if (!n)
95  chSequentialStreamPut(chp, '0');
96  else {
97  p = buf;
98  while (n)
99  *p++ = (n % 10) + '0', n /= 10;
100  while (p > buf)
101  chSequentialStreamPut(chp, *--p);
102  }
103 }

172 bool_t _test_assert_sequence(unsigned point, char *expected) {
173  char *cp = tokens_buffer;
174  while (cp < tokp) {
175  if (*cp++ != *expected++)
176  return _test_fail(point);

282 static void execute_test(const struct testcase *tcp) {

296 
297  test_wait_threads();

308 /**
309  * @brief Test execution thread function.
310  *
311  * @param[in] p pointer to a @p BaseChannel object for test output
312  * @return A failure boolean value.
313  */
314 msg_t TestThread(void *p) {
315  int i, j;
316 
317  chp = p;
318  test_println("");
319  test_println("*** ChibiOS/RT test suite");
320  test_println("***");
321  test_print("*** Kernel: ");
322  test_println(CH_KERNEL_VERSION);
323  test_print("*** Compiled: ");
324  test_println(__DATE__ " - " __TIME__);
325 #ifdef CH_COMPILER_NAME
326  test_print("*** Compiler: ");
327  test_println(CH_COMPILER_NAME);
328 #endif
329  test_print("*** Architecture: ");
330  test_println(CH_ARCHITECTURE_NAME);
331 #ifdef CH_CORE_VARIANT_NAME
332  test_print("*** Core Variant: ");
333  test_println(CH_CORE_VARIANT_NAME);
334 #endif
335 #ifdef CH_PORT_INFO
336  test_print("*** Port Info: ");
337  test_println(CH_PORT_INFO);
338 #endif
339 #ifdef PLATFORM_NAME
340  test_print("*** Platform: ");
341  test_println(PLATFORM_NAME);
342 #endif
343 #ifdef BOARD_NAME
344  test_print("*** Test Board: ");
345  test_println(BOARD_NAME);
346 #endif
347  test_println("");
348 
349  global_fail = FALSE;
350  i = 0;
351  while (patterns[i]) {
352  j = 0;
353  while (patterns[i][j]) {
354  print_line();
355  test_print("--- Test Case ");
356  test_printn(i + 1);
357  test_print(".");
358  test_printn(j + 1);
359  test_print(" (");
360  test_print(patterns[i][j]->name);
361  test_println(")");
362 #if DELAY_BETWEEN_TESTS > 0
363  chThdSleepMilliseconds(DELAY_BETWEEN_TESTS);
364 #endif
365  execute_test(patterns[i][j]);
366  if (local_fail) {
367  test_print("--- Result: FAILURE (#");
368  test_printn(failpoint);
369  test_print(" [");
370  print_tokens();
371  test_println("])");
372  }
373  else
374  test_println("--- Result: SUCCESS");
375  j++;
376  }
377  i++;
378  }
379  print_line();
380  test_println("");
381  test_print("Final result: ");
382  if (global_fail)
383  test_println("FAILURE");
384  else
385  test_println("SUCCESS");
386 
387  return (msg_t)global_fail;
388 }
