62 /**
63  * @page test_events_001 Events registration and dispatch
64  *
65  * <h2>Description</h2>
66  * Two event listeners are registered on an event source and then unregistered
67  * in the same order.<br>
68  * The test expects that the even source has listeners after the registrations
69  * and after the first unregistration, then, after the second unegistration,
70  * the test expects no more listeners.<br>
71  * In the second part the test dispatches three event flags and verifies that
72  * the associated event handlers are invoked in LSb-first order.
73  */

89  * Testing chEvtRegisterMask() and chEvtUnregister().

95  chEvtUnregister(&es1, &el1);

97  chEvtUnregister(&es1, &el2);
98  test_assert(3, !chEvtIsListeningI(&es1), "stuck listener");

101  * Testing chEvtDispatch().
102  */
103  chEvtDispatch(evhndl, 7);
104  test_assert_sequence(4, "ABC");
105 }

149 static void evt2_execute(void) {
150  eventmask_t m;
151  EventListener el1, el2;
152  systime_t target_time;
153 
154  /*
155  * Test on chEvtWaitOne() without wait.
156  */
157  chEvtAddEvents(7);
158  m = chEvtWaitOne(ALL_EVENTS);
159  test_assert(1, m == 1, "single event error");
160  m = chEvtWaitOne(ALL_EVENTS);
161  test_assert(2, m == 2, "single event error");
162  m = chEvtWaitOne(ALL_EVENTS);
163  test_assert(3, m == 4, "single event error");
164  m = chEvtGetAndClearEvents(ALL_EVENTS);
165  test_assert(4, m == 0, "stuck event");
166 
167  /*
168  * Test on chEvtWaitOne() with wait.
169  */
170  test_wait_tick();
171  target_time = chTimeNow() + MS2ST(50);
172  threads[0] = chThdCreateStatic(wa[0], WA_SIZE, chThdGetPriority() - 1,
173  thread1, chThdSelf());
174  m = chEvtWaitOne(ALL_EVENTS);
175  test_assert_time_window(5, target_time, target_time + ALLOWED_DELAY);
176  test_assert(6, m == 1, "single event error");
177  m = chEvtGetAndClearEvents(ALL_EVENTS);
178  test_assert(7, m == 0, "stuck event");
179  test_wait_threads();
180 
181  /*
182  * Test on chEvtWaitAny() without wait.
183  */
184  chEvtAddEvents(5);
185  m = chEvtWaitAny(ALL_EVENTS);
186  test_assert(8, m == 5, "unexpected pending bit");
187  m = chEvtGetAndClearEvents(ALL_EVENTS);
188  test_assert(9, m == 0, "stuck event");
189 
190  /*
191  * Test on chEvtWaitAny() with wait.
192  */
193  test_wait_tick();
194  target_time = chTimeNow() + MS2ST(50);
195  threads[0] = chThdCreateStatic(wa[0], WA_SIZE, chThdGetPriority() - 1,
196  thread1, chThdSelf());
197  m = chEvtWaitAny(ALL_EVENTS);
198  test_assert_time_window(10, target_time, target_time + ALLOWED_DELAY);
199  test_assert(11, m == 1, "single event error");
200  m = chEvtGetAndClearEvents(ALL_EVENTS);
201  test_assert(12, m == 0, "stuck event");
202  test_wait_threads();
203 
204  /*
205  * Test on chEvtWaitAll().
206  */
207  chEvtInit(&es1);
208  chEvtInit(&es2);
209  chEvtRegisterMask(&es1, &el1, 1);
210  chEvtRegisterMask(&es2, &el2, 4);
211  test_wait_tick();
212  target_time = chTimeNow() + MS2ST(50);
213  threads[0] = chThdCreateStatic(wa[0], WA_SIZE, chThdGetPriority() - 1,
214  thread2, "A");
215  m = chEvtWaitAll(5);
216  test_assert_time_window(13, target_time, target_time + ALLOWED_DELAY);
217  m = chEvtGetAndClearEvents(ALL_EVENTS);
218  test_assert(14, m == 0, "stuck event");
219  test_wait_threads();
220  chEvtUnregister(&es1, &el1);
221  chEvtUnregister(&es2, &el2);
222  test_assert(15, !chEvtIsListeningI(&es1), "stuck listener");
223  test_assert(16, !chEvtIsListeningI(&es2), "stuck listener");
224 }
