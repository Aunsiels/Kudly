97 
98  /* Do whatever else is needed to initialize interface. */
99 }

101 /*
102  * Transmits a frame.
103  */
104 static err_t low_level_output(struct netif *netif, struct pbuf *p) {
105  struct pbuf *q;
106  MACTransmitDescriptor td;
107 
108  (void)netif;
109  if (macWaitTransmitDescriptor(&ETHD1, &td, MS2ST(LWIP_SEND_TIMEOUT)) != RDY_OK)
110  return ERR_TIMEOUT;
111 
112 #if ETH_PAD_SIZE
113  pbuf_header(p, -ETH_PAD_SIZE); /* drop the padding word */
114 #endif
115 
116  /* Iterates through the pbuf chain. */
117  for(q = p; q != NULL; q = q->next)
118  macWriteTransmitDescriptor(&td, (uint8_t *)q->payload, (size_t)q->len);
119  macReleaseTransmitDescriptor(&td);
120 
121 #if ETH_PAD_SIZE
122  pbuf_header(p, ETH_PAD_SIZE); /* reclaim the padding word */
123 #endif
124 
125  LINK_STATS_INC(link.xmit);
126 
127  return ERR_OK;
128 }

130 /*
131  * Receives a frame.
132  */
133 static struct pbuf *low_level_input(struct netif *netif) {
134  MACReceiveDescriptor rd;
135  struct pbuf *p, *q;
136  u16_t len;
137 
138  (void)netif;
139  if (macWaitReceiveDescriptor(&ETHD1, &rd, TIME_IMMEDIATE) == RDY_OK) {
140  len = (u16_t)rd.size;
141 
142 #if ETH_PAD_SIZE
143  len += ETH_PAD_SIZE; /* allow room for Ethernet padding */
144 #endif
145 
146  /* We allocate a pbuf chain of pbufs from the pool. */
147  p = pbuf_alloc(PBUF_RAW, len, PBUF_POOL);
148 
149  if (p != NULL) {
150 
151 #if ETH_PAD_SIZE
152  pbuf_header(p, -ETH_PAD_SIZE); /* drop the padding word */
153 #endif
154 
155  /* Iterates through the pbuf chain. */
156  for(q = p; q != NULL; q = q->next)
157  macReadReceiveDescriptor(&rd, (uint8_t *)q->payload, (size_t)q->len);
158  macReleaseReceiveDescriptor(&rd);
159 
160 #if ETH_PAD_SIZE
161  pbuf_header(p, ETH_PAD_SIZE); /* reclaim the padding word */
162 #endif
163 
164  LINK_STATS_INC(link.recv);
165  }
166  else {
167  macReleaseReceiveDescriptor(&rd);
168  LINK_STATS_INC(link.memerr);
169  LINK_STATS_INC(link.drop);
170  }
171  return p;
172  }
173  return NULL;
174 }

176 /*
177  * Initialization.
178  */
179 static err_t ethernetif_init(struct netif *netif) {
180 #if LWIP_NETIF_HOSTNAME
181  /* Initialize interface hostname */
182  netif->hostname = "lwip";
183 #endif /* LWIP_NETIF_HOSTNAME */
184 
185  /*
186  * Initialize the snmp variables and counters inside the struct netif.
187  * The last argument should be replaced with your link speed, in units
188  * of bits per second.
189  */
190  NETIF_INIT_SNMP(netif, snmp_ifType_ethernet_csmacd, LWIP_LINK_SPEED);
191 
192  netif->state = NULL;
193  netif->name[0] = LWIP_IFNAME0;
194  netif->name[1] = LWIP_IFNAME1;
195  /* We directly use etharp_output() here to save a function call.
196  * You can instead declare your own function an call etharp_output()
197  * from it if you have to do some checks before sending (e.g. if link
198  * is available...) */
199  netif->output = etharp_output;
200  netif->linkoutput = low_level_output;
201 
202  /* initialize the hardware */
203  low_level_init(netif);
204 
205  return ERR_OK;
206 }

208 /**
209  * @brief LWIP handling thread.
210  *
211  * @param[in] p pointer to a @p lwipthread_opts structure or @p NULL
212  * @return The function does not return.
213  */
214 msg_t lwip_thread(void *p) {
215  EvTimer evt;
216  EventListener el0, el1;
217  struct ip_addr ip, gateway, netmask;
218  static struct netif thisif;
219  static const MACConfig mac_config = {thisif.hwaddr};
220 
221  chRegSetThreadName("lwipthread");
222 
223  /* Initializes the thing.*/
224  tcpip_init(NULL, NULL);
225 
226  /* TCP/IP parameters, runtime or compile time.*/
227  if (p) {
228  struct lwipthread_opts *opts = p;
229  unsigned i;
230 
231  for (i = 0; i < 6; i++)
232  thisif.hwaddr[i] = opts->macaddress[i];
233  ip.addr = opts->address;
234  gateway.addr = opts->gateway;
235  netmask.addr = opts->netmask;
236  }
237  else {
238  thisif.hwaddr[0] = LWIP_ETHADDR_0;
239  thisif.hwaddr[1] = LWIP_ETHADDR_1;
240  thisif.hwaddr[2] = LWIP_ETHADDR_2;
241  thisif.hwaddr[3] = LWIP_ETHADDR_3;
242  thisif.hwaddr[4] = LWIP_ETHADDR_4;
243  thisif.hwaddr[5] = LWIP_ETHADDR_5;
244  LWIP_IPADDR(&ip);
245  LWIP_GATEWAY(&gateway);
246  LWIP_NETMASK(&netmask);
247  }
248  macStart(&ETHD1, &mac_config);
249  netif_add(&thisif, &ip, &netmask, &gateway, NULL, ethernetif_init, tcpip_input);
250 
251  netif_set_default(&thisif);
252  netif_set_up(&thisif);
253 
254  /* Setup event sources.*/
255  evtInit(&evt, LWIP_LINK_POLL_INTERVAL);
256  evtStart(&evt);
257  chEvtRegisterMask(&evt.et_es, &el0, PERIODIC_TIMER_ID);
258  chEvtRegisterMask(macGetReceiveEventSource(&ETHD1), &el1, FRAME_RECEIVED_ID);
259  chEvtAddEvents(PERIODIC_TIMER_ID | FRAME_RECEIVED_ID);
260 
261  /* Goes to the final priority after initialization.*/
262  chThdSetPriority(LWIP_THREAD_PRIORITY);
263 
264  while (TRUE) {
265  eventmask_t mask = chEvtWaitAny(ALL_EVENTS);
266  if (mask & PERIODIC_TIMER_ID) {
267  bool_t current_link_status = macPollLinkStatus(&ETHD1);
268  if (current_link_status != netif_is_link_up(&thisif)) {
269  if (current_link_status)
270  tcpip_callback_with_block((tcpip_callback_fn) netif_set_link_up,
271  &thisif, 0);
272  else
273  tcpip_callback_with_block((tcpip_callback_fn) netif_set_link_down,
274  &thisif, 0);
275  }
276  }
277  if (mask & FRAME_RECEIVED_ID) {
278  struct pbuf *p;
279  while ((p = low_level_input(&thisif)) != NULL) {
280  struct eth_hdr *ethhdr = p->payload;
281  switch (htons(ethhdr->type)) {
282  /* IP or ARP packet? */
283  case ETHTYPE_IP:
284  case ETHTYPE_ARP:
285 #if PPPOE_SUPPORT
286  /* PPPoE packet? */
287  case ETHTYPE_PPPOEDISC:
288  case ETHTYPE_PPPOE:
289 #endif /* PPPOE_SUPPORT */
290  /* full packet send to tcpip_thread to process */
291  if (thisif.input(p, &thisif) == ERR_OK)
292  break;
293  LWIP_DEBUGF(NETIF_DEBUG, ("ethernetif_input: IP input error\n"));
294  default:
295  pbuf_free(p);
296  }
297  }
298  }
299  }
300  return 0;
301 }
