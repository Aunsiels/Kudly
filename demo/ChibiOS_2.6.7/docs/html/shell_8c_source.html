37 static char *_strtok(char *str, const char *delim, char **saveptr) {
38  char *token;
39  if (str)
40  *saveptr = str;
41  token = *saveptr;
42 
43  if (!token)
44  return NULL;
45 
46  token += strspn(token, delim);
47  *saveptr = strpbrk(token, delim);
48  if (*saveptr)
49  *(*saveptr)++ = '\0';
50 
51  return *token ? token : NULL;
52 }

131 /**
132  * @brief Shell thread function.
133  *
134  * @param[in] p pointer to a @p BaseSequentialStream object
135  * @return Termination reason.
136  * @retval RDY_OK terminated by command.
137  * @retval RDY_RESET terminated by reset condition on the I/O channel.
138  *
139  * @notapi
140  */
141 static msg_t shell_thread(void *p) {
142  int n;
143  BaseSequentialStream *chp = ((ShellConfig *)p)->sc_channel;
144  const ShellCommand *scp = ((ShellConfig *)p)->sc_commands;
145  char *lp, *cmd, *tokp, line[SHELL_MAX_LINE_LENGTH];
146  char *args[SHELL_MAX_ARGUMENTS + 1];
147 
148  chRegSetThreadName("shell");
149  chprintf(chp, "\r\nChibiOS/RT Shell\r\n");
150  while (TRUE) {
151  chprintf(chp, "ch> ");
152  if (shellGetLine(chp, line, sizeof(line))) {
153  chprintf(chp, "\r\nlogout");
154  break;
155  }
156  lp = _strtok(line, " \t", &tokp);
157  cmd = lp;
158  n = 0;
159  while ((lp = _strtok(NULL, " \t", &tokp)) != NULL) {
160  if (n >= SHELL_MAX_ARGUMENTS) {
161  chprintf(chp, "too many arguments\r\n");
162  cmd = NULL;
163  break;
164  }
165  args[n++] = lp;
166  }
167  args[n] = NULL;
168  if (cmd != NULL) {
169  if (strcasecmp(cmd, "exit") == 0) {
170  if (n > 0) {
171  usage(chp, "exit");
172  continue;
173  }
174  break;
175  }
176  else if (strcasecmp(cmd, "help") == 0) {
177  if (n > 0) {
178  usage(chp, "help");
179  continue;
180  }
181  chprintf(chp, "Commands: help exit ");
182  list_commands(chp, local_commands);
183  if (scp != NULL)
184  list_commands(chp, scp);
185  chprintf(chp, "\r\n");
186  }
187  else if (cmdexec(local_commands, chp, cmd, n, args) &&
188  ((scp == NULL) || cmdexec(scp, chp, cmd, n, args))) {
189  chprintf(chp, "%s", cmd);
190  chprintf(chp, " ?\r\n");
191  }
192  }
193  }
194  shellExit(RDY_OK);
195  /* Never executed, silencing a warning.*/
196  return 0;
197 }

263 /**
264  * @brief Reads a whole line from the input channel.
265  *
266  * @param[in] chp pointer to a @p BaseSequentialStream object
267  * @param[in] line pointer to the line buffer
268  * @param[in] size buffer maximum length
269  * @return The operation status.
270  * @retval TRUE the channel was reset or CTRL-D pressed.
271  * @retval FALSE operation successful.
272  *
273  * @api
274  */
275 bool_t shellGetLine(BaseSequentialStream *chp, char *line, unsigned size) {
276  char *p = line;
277 
278  while (TRUE) {
279  char c;
280 
281  if (chSequentialStreamRead(chp, (uint8_t *)&c, 1) == 0)
282  return TRUE;
283  if (c == 4) {
284  chprintf(chp, "^D");
285  return TRUE;
286  }
287  if ((c == 8) || (c == 127)) {
288  if (p != line) {
289  chSequentialStreamPut(chp, c);
290  chSequentialStreamPut(chp, 0x20);
291  chSequentialStreamPut(chp, c);
292  p--;
293  }
294  continue;
295  }
296  if (c == '\r') {
297  chprintf(chp, "\r\n");
298  *p = 0;
299  return FALSE;
300  }
301  if (c < 0x20)
302  continue;
303  if (p < line + size - 1) {
304  chSequentialStreamPut(chp, c);
305  *p++ = (char)c;
306  }
307  }
308 }
