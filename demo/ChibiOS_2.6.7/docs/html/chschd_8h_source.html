214 /**
215  * @brief Inline-able preemption code.
216  * @details This is the common preemption code, this function must be invoked
217  * exclusively from the port layer.
218  *
219  * @special
220  */
221 #if (CH_TIME_QUANTUM > 0) || defined(__DOXYGEN__)
222 #define chSchPreemption() { \
223  tprio_t p1 = firstprio(&rlist.r_queue); \
224  tprio_t p2 = currp->p_prio; \
225  if (currp->p_preempt) { \
226  if (p1 > p2) \
227  chSchDoRescheduleAhead(); \
228  } \
229  else { \
230  if (p1 >= p2) \
231  chSchDoRescheduleBehind(); \
232  } \
233 }
234 #else /* CH_TIME_QUANTUM == 0 */
235 #define chSchPreemption() { \
236  if (p1 >= p2) \
237  chSchDoRescheduleAhead(); \
238 }
239 #endif /* CH_TIME_QUANTUM == 0 */
