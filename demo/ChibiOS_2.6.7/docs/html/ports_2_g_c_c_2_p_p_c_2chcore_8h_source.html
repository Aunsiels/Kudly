242 /**
243  * @brief Platform dependent part of the @p chThdCreateI() API.
244  * @details This code usually setup the context switching frame represented
245  * by an @p intctx structure.
246  */
247 #define SETUP_CONTEXT(workspace, wsize, pf, arg) { \
248  uint8_t *sp = (uint8_t *)workspace + wsize - sizeof(struct eabi_frame); \
249  ((struct eabi_frame *)sp)->slink = 0; \
250  ((struct eabi_frame *)sp)->shole = _port_thread_start; \
251  tp->p_ctx.sp = (struct intctx *)(sp - sizeof(struct intctx)); \
252  tp->p_ctx.sp->r31 = arg; \
253  tp->p_ctx.sp->r30 = pf; \
254 }

352 /**
353  * @brief Performs a context switch between two threads.
354  * @details This is the most critical code in any port, this function
355  * is responsible for the context switch between 2 threads.
356  * @note The implementation of this code affects <b>directly</b> the context
357  * switch performance so optimize here as much as you can.
358  *
359  * @param[in] ntp the thread to be switched in
360  * @param[in] otp the thread to be switched out
361  */
362 #if !CH_DBG_ENABLE_STACK_CHECK || defined(__DOXYGEN__)
363 #define port_switch(ntp, otp) _port_switch(ntp, otp)
364 #else
365 #define port_switch(ntp, otp) { \
366  register struct intctx *sp asm ("%r1"); \
367  if ((stkalign_t *)(sp - 1) < otp->p_stklimit) \
368  chDbgPanic("stack overflow"); \
369  _port_switch(ntp, otp); \
370 }
371 #endif
