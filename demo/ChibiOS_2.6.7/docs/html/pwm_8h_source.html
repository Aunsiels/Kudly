109 /**
110  * @name PWM duty cycle conversion
111  * @{
112  */
113 /**
114  * @brief Converts from fraction to pulse width.
115  * @note Be careful with rounding errors, this is integer math not magic.
116  * You can specify tenths of thousandth but make sure you have the
117  * proper hardware resolution by carefully choosing the clock source
118  * and prescaler settings, see @p PWM_COMPUTE_PSC.
119  *
120  * @param[in] pwmp pointer to a @p PWMDriver object
121  * @param[in] denominator denominator of the fraction
122  * @param[in] numerator numerator of the fraction
123  * @return The pulse width to be passed to @p pwmEnableChannel().
124  *
125  * @api
126  */
127 #define PWM_FRACTION_TO_WIDTH(pwmp, denominator, numerator) \
128  ((pwmcnt_t)((((pwmcnt_t)(pwmp)->period) * \
129  (pwmcnt_t)(numerator)) / (pwmcnt_t)(denominator)))
130 
131 /**

136  * and prescaler settings, see @p PWM_COMPUTE_PSC.
137  *

168 /**
169  * @brief Changes the period the PWM peripheral.
170  * @details This function changes the period of a PWM unit that has already
171  * been activated using @p pwmStart().
172  * @pre The PWM unit must have been activated using @p pwmStart().
173  * @post The PWM unit period is changed to the new value.
174  * @note If a period is specified that is shorter than the pulse width
175  * programmed in one of the channels then the behavior is not
176  * guaranteed.
177  *
178  * @param[in] pwmp pointer to a @p PWMDriver object
179  * @param[in] value new cycle time in ticks
180  *
181  * @iclass
182  */
183 #define pwmChangePeriodI(pwmp, value) { \
184  (pwmp)->period = (value); \
185  pwm_lld_change_period(pwmp, value); \
186 }
