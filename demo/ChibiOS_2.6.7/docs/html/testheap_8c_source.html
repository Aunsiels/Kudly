69 static void heap1_execute(void) {
70  void *p1, *p2, *p3;
71  size_t n, sz;
72 
73  /* Unrelated, for coverage only.*/
74  (void)chCoreStatus();
75 
76  /*
77  * Test on the default heap in order to cover the core allocator at
78  * least one time.
79  */
80  (void)chHeapStatus(NULL, &sz);
81  p1 = chHeapAlloc(NULL, SIZE);
82  test_assert(1, p1 != NULL, "allocation failed");
83  chHeapFree(p1);
84  p1 = chHeapAlloc(NULL, (size_t)-256);
85  test_assert(2, p1 == NULL, "allocation not failed");
86 
87  /* Initial local heap state.*/
88  (void)chHeapStatus(&test_heap, &sz);
89 
90  /* Same order.*/
91  p1 = chHeapAlloc(&test_heap, SIZE);
92  p2 = chHeapAlloc(&test_heap, SIZE);
93  p3 = chHeapAlloc(&test_heap, SIZE);
94  chHeapFree(p1); /* Does not merge.*/
95  chHeapFree(p2); /* Merges backward.*/
96  chHeapFree(p3); /* Merges both sides.*/
97  test_assert(3, chHeapStatus(&test_heap, &n) == 1, "heap fragmented");
98 
99  /* Reverse order.*/
100  p1 = chHeapAlloc(&test_heap, SIZE);
101  p2 = chHeapAlloc(&test_heap, SIZE);
102  p3 = chHeapAlloc(&test_heap, SIZE);
103  chHeapFree(p3); /* Merges forward.*/
104  chHeapFree(p2); /* Merges forward.*/
105  chHeapFree(p1); /* Merges forward.*/
106  test_assert(4, chHeapStatus(&test_heap, &n) == 1, "heap fragmented");
107 
108  /* Small fragments handling.*/
109  p1 = chHeapAlloc(&test_heap, SIZE + 1);
110  p2 = chHeapAlloc(&test_heap, SIZE);
111  chHeapFree(p1);
112  test_assert(5, chHeapStatus(&test_heap, &n) == 2, "invalid state");
113  p1 = chHeapAlloc(&test_heap, SIZE);
114  /* Note, the first situation happens when the alignment size is smaller
115  than the header size, the second in the other cases.*/
116  test_assert(6, (chHeapStatus(&test_heap, &n) == 1) ||
117  (chHeapStatus(&test_heap, &n) == 2), "heap fragmented");
118  chHeapFree(p2);
119  chHeapFree(p1);
120  test_assert(7, chHeapStatus(&test_heap, &n) == 1, "heap fragmented");
121 
122  /* Skip fragment handling.*/
123  p1 = chHeapAlloc(&test_heap, SIZE);
124  p2 = chHeapAlloc(&test_heap, SIZE);
125  chHeapFree(p1);
126  test_assert(8, chHeapStatus(&test_heap, &n) == 2, "invalid state");
127  p1 = chHeapAlloc(&test_heap, SIZE * 2); /* Skips first fragment.*/
128  chHeapFree(p1);
129  chHeapFree(p2);
130  test_assert(9, chHeapStatus(&test_heap, &n) == 1, "heap fragmented");
131 
132  /* Allocate all handling.*/
133  (void)chHeapStatus(&test_heap, &n);
134  p1 = chHeapAlloc(&test_heap, n);
135  test_assert(10, chHeapStatus(&test_heap, &n) == 0, "not empty");
136  chHeapFree(p1);
137 
138  test_assert(11, chHeapStatus(&test_heap, &n) == 1, "heap fragmented");
139  test_assert(12, n == sz, "size changed");
140 }
