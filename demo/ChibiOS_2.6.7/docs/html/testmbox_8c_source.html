76 static void mbox1_execute(void) {
77  msg_t msg1, msg2;
78  unsigned i;
79 
80  /*
81  * Testing initial space.
82  */
83  test_assert(1, chMBGetFreeCountI(&mb1) == MB_SIZE, "wrong size");
84 
85  /*
86  * Testing enqueuing and backward circularity.
87  */
88  for (i = 0; i < MB_SIZE - 1; i++) {
89  msg1 = chMBPost(&mb1, 'B' + i, TIME_INFINITE);
90  test_assert(2, msg1 == RDY_OK, "wrong wake-up message");
91  }
92  msg1 = chMBPostAhead(&mb1, 'A', TIME_INFINITE);
93  test_assert(3, msg1 == RDY_OK, "wrong wake-up message");
94 
95  /*
96  * Testing post timeout.
97  */
98  msg1 = chMBPost(&mb1, 'X', 1);
99  test_assert(4, msg1 == RDY_TIMEOUT, "wrong wake-up message");
100  chSysLock();
101  msg1 = chMBPostI(&mb1, 'X');
102  chSysUnlock();
103  test_assert(5, msg1 == RDY_TIMEOUT, "wrong wake-up message");
104  msg1 = chMBPostAhead(&mb1, 'X', 1);
105  test_assert(6, msg1 == RDY_TIMEOUT, "wrong wake-up message");
106  chSysLock();
107  msg1 = chMBPostAheadI(&mb1, 'X');
108  chSysUnlock();
109  test_assert(7, msg1 == RDY_TIMEOUT, "wrong wake-up message");
110 
111  /*
112  * Testing final conditions.
113  */
114  test_assert_lock(8, chMBGetFreeCountI(&mb1) == 0, "still empty");
115  test_assert_lock(9, chMBGetUsedCountI(&mb1) == MB_SIZE, "not full");
116  test_assert_lock(10, mb1.mb_rdptr == mb1.mb_wrptr, "pointers not aligned");
117 
118  /*
119  * Testing dequeuing.
120  */
121  for (i = 0; i < MB_SIZE; i++) {
122  msg1 = chMBFetch(&mb1, &msg2, TIME_INFINITE);
123  test_assert(11, msg1 == RDY_OK, "wrong wake-up message");
124  test_emit_token(msg2);
125  }
126  test_assert_sequence(12, "ABCDE");
127 
128  /*
129  * Testing buffer circularity.
130  */
131  msg1 = chMBPost(&mb1, 'B' + i, TIME_INFINITE);
132  test_assert(13, msg1 == RDY_OK, "wrong wake-up message");
133  msg1 = chMBFetch(&mb1, &msg2, TIME_INFINITE);
134  test_assert(14, msg1 == RDY_OK, "wrong wake-up message");
135  test_assert(15, mb1.mb_buffer == mb1.mb_wrptr, "write pointer not aligned to base");
136  test_assert(16, mb1.mb_buffer == mb1.mb_rdptr, "read pointer not aligned to base");
137 
138  /*
139  * Testing fetch timeout.
140  */
141  msg1 = chMBFetch(&mb1, &msg2, 1);
142  test_assert(17, msg1 == RDY_TIMEOUT, "wrong wake-up message");
143  chSysLock();
144  msg1 = chMBFetchI(&mb1, &msg2);
145  chSysUnlock();
146  test_assert(18, msg1 == RDY_TIMEOUT, "wrong wake-up message");
147 
148  /*
149  * Testing final conditions.
150  */
151  test_assert_lock(19, chMBGetFreeCountI(&mb1) == MB_SIZE, "not empty");
152  test_assert_lock(20, chMBGetUsedCountI(&mb1) == 0, "still full");
153  test_assert_lock(21, mb1.mb_rdptr == mb1.mb_wrptr, "pointers not aligned");
154 
155  /*
156  * Testing I-Class.
157  */
158  chSysLock();
159  msg1 = chMBPostI(&mb1, 'A');
160  test_assert(22, msg1 == RDY_OK, "wrong wake-up message");
161  msg1 = chMBPostI(&mb1, 'B');
162  test_assert(23, msg1 == RDY_OK, "wrong wake-up message");
163  msg1 = chMBPostI(&mb1, 'C');
164  test_assert(24, msg1 == RDY_OK, "wrong wake-up message");
165  msg1 = chMBPostI(&mb1, 'D');
166  test_assert(25, msg1 == RDY_OK, "wrong wake-up message");
167  msg1 = chMBPostI(&mb1, 'E');
168  chSysUnlock();
169  test_assert(26, msg1 == RDY_OK, "wrong wake-up message");
170  test_assert(27, mb1.mb_rdptr == mb1.mb_wrptr, "pointers not aligned");
171  for (i = 0; i < MB_SIZE; i++) {
172  chSysLock();
173  msg1 = chMBFetchI(&mb1, &msg2);
174  chSysUnlock();
175  test_assert(28, msg1 == RDY_OK, "wrong wake-up message");
176  test_emit_token(msg2);
177  }
178  test_assert_sequence(29, "ABCDE");
179  test_assert_lock(30, chMBGetFreeCountI(&mb1) == MB_SIZE, "not empty");
180  test_assert_lock(31, chMBGetUsedCountI(&mb1) == 0, "still full");
181  test_assert(32, mb1.mb_rdptr == mb1.mb_wrptr, "pointers not aligned");
182
