28 /**
29  * @file io_block.h
30  * @brief I/O block devices access.
31  * @details This header defines an abstract interface useful to access generic
32  * I/O block devices in a standardized way.
33  *
34  * @addtogroup IO_BLOCK
35  * @details This module defines an abstract interface for accessing generic
36  * block devices.<br>
37  * Note that no code is present, just abstract interfaces-like
38  * structures, you should look at the system as to a set of
39  * abstract C++ classes (even if written in C). This system
40  * has then advantage to make the access to block devices
41  * independent from the implementation logic.
42  * @{
43  */

48 /**
49  * @brief Driver state machine possible states.
50  */
51 typedef enum {
52  BLK_UNINIT = 0, /**< Not initialized. */
53  BLK_STOP = 1, /**< Stopped. */
54  BLK_ACTIVE = 2, /**< Interface active. */
55  BLK_CONNECTING = 3, /**< Connection in progress. */
56  BLK_DISCONNECTING = 4, /**< Disconnection in progress. */
57  BLK_READY = 5, /**< Device ready. */
58  BLK_READING = 6, /**< Read operation in progress. */
59  BLK_WRITING = 7, /**< Write operation in progress. */
60  BLK_SYNCING = 8 /**< Sync. operation in progress. */
61 } blkstate_t;
