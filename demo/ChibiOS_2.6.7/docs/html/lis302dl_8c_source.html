52 /**
53  * @brief Reads a register value.
54  * @pre The SPI interface must be initialized and the driver started.
55  *
56  * @param[in] spip pointer to the SPI initerface
57  * @param[in] reg register number
58  * @return The register value.
59  */
60 uint8_t lis302dlReadRegister(SPIDriver *spip, uint8_t reg) {
61 
62  spiSelect(spip);
63  txbuf[0] = 0x80 | reg;
64  txbuf[1] = 0xff;
65  spiExchange(spip, 2, txbuf, rxbuf);
66  spiUnselect(spip);
67  return rxbuf[1];
68 }

70 /**
71  * @brief Writes a value into a register.
72  * @pre The SPI interface must be initialized and the driver started.
73  *
74  * @param[in] spip pointer to the SPI initerface
75  * @param[in] reg register number
76  * @param[in] value the value to be written
77  */
78 void lis302dlWriteRegister(SPIDriver *spip, uint8_t reg, uint8_t value) {
79 
80  switch (reg) {
81  default:
82  /* Reserved register must not be written, according to the datasheet
83  this could permanently damage the device.*/
84  chDbgAssert(FALSE, "lis302dlWriteRegister(), #1", "reserved register");
85  case LIS302DL_WHO_AM_I:
86  case LIS302DL_HP_FILTER_RESET:
87  case LIS302DL_STATUS_REG:
88  case LIS302DL_OUTX:
89  case LIS302DL_OUTY:
90  case LIS302DL_OUTZ:
91  case LIS302DL_FF_WU_SRC1:
92  case LIS302DL_FF_WU_SRC2:
93  case LIS302DL_CLICK_SRC:
94  /* Read only registers cannot be written, the command is ignored.*/
95  return;
96  case LIS302DL_CTRL_REG1:
97  case LIS302DL_CTRL_REG2:
98  case LIS302DL_CTRL_REG3:
99  case LIS302DL_FF_WU_CFG1:
100  case LIS302DL_FF_WU_THS1:
101  case LIS302DL_FF_WU_DURATION1:
102  case LIS302DL_FF_WU_CFG2:
103  case LIS302DL_FF_WU_THS2:
104  case LIS302DL_FF_WU_DURATION2:
105  case LIS302DL_CLICK_CFG:
106  case LIS302DL_CLICK_THSY_X:
107  case LIS302DL_CLICK_THSZ:
108  case LIS302DL_CLICK_TIMELIMIT:
109  case LIS302DL_CLICK_LATENCY:
110  case LIS302DL_CLICK_WINDOW:
111  spiSelect(spip);
112  txbuf[0] = reg;
113  txbuf[1] = value;
114  spiSend(spip, 2, txbuf);
115  spiUnselect(spip);
116  }
117 }
